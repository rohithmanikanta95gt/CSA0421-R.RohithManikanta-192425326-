#include <stdio.h>
#include <unistd.h>

int main() {
    pid_t pid;

    pid = fork();   // create a new process

    if (pid == 0) {
        // Child process
        printf("Child Process: PID = %d, PPID = %d\n", getpid(), getppid());
    }
    else if (pid > 0) {
        // Parent process
        printf("Parent Process: PID = %d, PPID = %d\n", getpid(), getppid());
    }
    else {
        // Fork failed
        printf("Fork failed\n");
    }

    return 0;
}
